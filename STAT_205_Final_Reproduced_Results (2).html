def find_trajectory_length(xdata,ydata,zdata):
    length = 0
    for i in range(0,10000):
        #print(xdata[i], ydata[i],zdata[i])
        add_me = ((xdata[i] -xdata[i-1])**2 + (ydata[i] -ydata[i-1])**2+ (zdata[i] -zdata[i-1])**2)**1/2
        length = length+add_me
    return length
